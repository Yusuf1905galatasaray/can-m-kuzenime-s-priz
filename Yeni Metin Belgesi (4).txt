    <script>
        const audio = document.getElementById('birthday-audio');
        const musicBtn = document.getElementById('music-btn');
        const musicText = document.getElementById('music-text');
        
        let isPlaying = false;
        let playPromise = null; // Çakışmaları önlemek için promise takibi

        function startCelebration() {
            document.getElementById('start-screen').classList.add('hidden');
            const screen = document.getElementById('celebration-screen');
            screen.classList.remove('hidden');
            
            launchPremiumConfetti();
            createMagicElements();
            
            // İlk başlatma
            safePlay();
        }

        // Güvenli çalma fonksiyonu (Hata önleyici)
        function safePlay() {
            playPromise = audio.play();
            
            if (playPromise !== undefined) {
                playPromise.then(() => {
                    isPlaying = true;
                    musicText.innerText = "Müziği Durdur";
                }).catch(error => {
                    console.log("Oynatma durduruldu veya engellendi.");
                    isPlaying = false;
                    musicText.innerText = "Müziği Başlat";
                });
            }
        }

        // Güvenli durdurma fonksiyonu
        function safePause() {
            if (playPromise !== undefined) {
                playPromise.then(() => {
                    audio.pause();
                    isPlaying = false;
                    musicText.innerText = "Müziği Başlat";
                });
            } else {
                audio.pause();
                isPlaying = false;
                musicText.innerText = "Müziği Başlat";
            }
        }

        function handleMusicToggle() {
            if (isPlaying) {
                safePause();
            } else {
                safePlay();
            }
        }

        function launchPremiumConfetti() {
            const count = 250;
            const defaults = { origin: { y: 0.6 }, zIndex: 1000 };

            function fire(particleRatio, opts) {
                confetti(Object.assign({}, defaults, opts, {
                    particleCount: Math.floor(count * particleRatio)
                }));
            }

            fire(0.25, { spread: 26, startVelocity: 55 });
            fire(0.2, { spread: 60 });
            fire(0.35, { spread: 100, decay: 0.91, scalar: 0.8 });
            fire(0.1, { spread: 120, startVelocity: 25, decay: 0.92, scalar: 1.2 });
            fire(0.1, { spread: 120, startVelocity: 45 });
        }

        function createMagicElements() {
            const elements = ['✨', '💖', '⭐', '🎈', '🍭'];
            setInterval(() => {
                const el = document.createElement('div');
                el.className = 'floating-element';
                el.innerText = elements[Math.floor(Math.random() * elements.length)];
                el.style.left = Math.random() * 100 + 'vw';
                el.style.fontSize = (Math.random() * 20 + 20) + 'px';
                el.style.setProperty('--d', (Math.random() * 5 + 5) + 's');
                document.body.appendChild(el);
                
                setTimeout(() => el.remove(), 10000);
            }, 600);
        }
